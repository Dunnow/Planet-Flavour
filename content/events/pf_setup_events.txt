namespace = pf_setup

event = {
	id = pf_setup.0
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	immediate = { set_global_flag = planet_flavour_is_active }
}

event = {
    id = pf_setup.10
	hide_window = yes
	is_triggered_only = yes

	immediate = {
        every_system = {
            limit = {
                NOT = {
                    has_star_flag = home_system
                }
                count_system_planet = {
                    limit = {
                        pf_is_valid_planet_for_extraplanetary_life = yes
                    }
                    count > 1
                }
            }
            every_system_planet = {
                limit = {
                    pf_is_valid_planet_for_extraplanetary_life = yes
                }
                add_modifier = { modifier = pf_pm_extraplanetary_life }
            }
        }
    }
}

event = {
    id = pf_setup.11
	hide_window = yes
	is_triggered_only = yes

	immediate = {
        every_system = {
            limit = {
                any_system_planet = {
                    has_ring = yes
                }
            }
            every_system_planet = {
                limit = { has_ring = yes }
                add_modifier = { modifier = pf_pm_planetary_rings }
            }
        }
    }
}

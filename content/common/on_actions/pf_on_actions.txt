on_game_start = {
    events = {
        pf_core.0      # Set mod flag
        pf_core.10     # Apply extraplanetary life modifier
        pf_core.11     # Apply planetary rings modifier
        pf_core.30     # Apply dry world entity
        pf_core.31     # Apply swamp world blockers
        pf_core.40     # Apply tomb world modifiers
        pf_core.43     # Apply nuclear winter to old foes system
        pf_misc.110    # Roll to spawn Hive Stars
        pf_core.200    # Distribute luxury homeworlds
        pf_core.100    # Clear invalid modifiers from planets
    }
}

on_planet_class_changed = {
    events = {
        pf_core.20      # Normal habitable class change
        pf_core.21      # Limited / No biosphere class changes
        pf_core.22      # Multi planetary ecosystem class change check
        pf_core.42      # Apply nuclear winter to pre-ftl nuclear exchanges
        pf_core.50      # Swap micro terraforming environmental enhancement deposits
    }
}

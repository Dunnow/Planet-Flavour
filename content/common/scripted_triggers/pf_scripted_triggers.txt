pf_is_valid_planet_for_extraplanetary_life = {
    pf_is_habitable_planet = yes
    is_artificial = no
    is_colony = no

    NOR = {
        is_planet_class = pc_nuked
        is_planet_class = pc_city
        is_planet_class = pc_relic
        is_planet_class = pc_hive
        is_planet_class = pc_machine
    }
}

pf_is_world_with_ecosystems = {
    pf_is_habitable_planet = yes
    is_artificial = no

    NOR = {
        is_planet_class = pc_nuked
        is_planet_class = pc_city
        is_planet_class = pc_relic
        is_planet_class = pc_hive
        is_planet_class = pc_machine
    }
}

pf_is_habitable_planet = {
    OR = {
        is_colonizable = yes
        is_colony = yes
    }
}
